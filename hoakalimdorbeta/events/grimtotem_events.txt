# Invitation to Spirithorns (from GRM_befriend_spirithorn_nf)
country_event = {
	id = grimtotem.2
	title = grimtotem.2.t
	desc = grimtotem.2.d

	is_triggered_only = yes

	fire_only_once = yes

	#They accept
	option = {
		name = grimtotem.2.option.1

		custom_effect_tooltip = grimtotem.2.option.1.d

		GRM = {
			country_event = {
				id = grimtotem.3

			}
		}
	}

	#They refuse
	option = {
		name = grimtotem.2.option.2

		GRM = {
			country_event = {
				id = grimtotem.4
				days = 1
			}
		}
	}
}

# Spirithorn alliance accepted (from grimtotem.2.option.1)
country_event = {
	id = grimtotem.3
	title = grimtotem.3.t
	desc = grimtotem.3.d

	is_triggered_only = yes

	fire_only_once = yes

	option = {
		name = grimtotem.3.option.1

		add_stability = 0.05

		if = {
			limit = {
				is_in_faction = yes
			}
			GRM = { add_to_faction = SPH }
		}
		else = {
			create_faction = "Tauren United Tribes"
			GRM = { add_to_faction = SPH }
		}
	}
}
