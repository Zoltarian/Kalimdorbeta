											size > 0

							}

					add_building_construction = {
							type = industrial_complex
							level = 1
							instant_build = yes
					}
				}
			}
			25 = {
				add_stability = 0.05
			}
			25 = {
				add_timed_idea = {
					idea =  council_advisor_bonus
					days = 90
				}	
			}
		}
	}
}
