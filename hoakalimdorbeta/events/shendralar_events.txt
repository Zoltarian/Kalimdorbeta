# Finding books monthly tick (from shendralar.19)
country_event = {
	id = shendralar.19
	title = shendralar.19.t
	desc = shendralar.19.d

	is_triggered_only = yes
	fire_only_once = no
	hidden = yes

	immediate = {
		random_list = {
			50 = {
				#nothing
			}
			50 = {
				add_to_variable = {
					var = SHD_finding_book_tick
					value = 1
				}
			}
		}

		#Effects happen everytime the variable gets to 3
		if = {
			limit = {
				check_variable = {
					var = SHD_finding_book_tick
					value = 3
					compare = equals
				}
			}
			#Reset variable
			set_variable = {
				var = SHD_finding_book_tick
				value = 0
			}
			#Set variable so the event get the right effect
			random_list = {
				34 = {
					set_variable = {
						var = SHD_book_effect
						value = administration_bonus
					}
				}
				33 = {
					set_variable = {
						var = SHD_book_effect
						value = doctrine_bonus
					}
				}
				33 = {
					set_variable = {
						var = SHD_book_effect
						value = troops_bonus
					}
				}
			}
			#Fire the visible event
			country_event = shendralar.20
		}
	}
}
