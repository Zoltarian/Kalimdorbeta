	accept_church_of_the_holy_light_decision = {

		available = {
			any_owned_state = {
				check_variable = { religion = 1 }
			}
			NOT = {
				check_variable = { accepted_religions^num > 4 }
				has_country_flag = accepting_a_religion
			}
		}

		visible = {
			AND = {
				any_owned_state = {
					check_variable = { religion = 1 }
				}
				OR = {
					NOT = {
						has_idea = primary_religion_the_church_of_the_holy_light
						has_idea = secondary_religion_the_church_of_the_holy_light
						check_variable = { accepted_religions^num > 4 }
					}
				}
			}
		}

		fire_only_once = no

		cost = 300
		days_remove = 365
		days_re_enable = 365

		remove_effect = {
			add_ideas = {
				secondary_religion_the_church_of_the_holy_light
			}
			add_to_array = {
				array = accepted_religions
				value = 1
			}
			clr_country_flag = accepting_a_religion
		}

		complete_effect = {
			set_country_flag = accepting_a_religion
		}
	}
