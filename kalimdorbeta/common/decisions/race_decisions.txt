	accept_ancients_decision = {

		available = {
			any_owned_state = {
				check_variable = { race = 1 }
			}
			NOT = {
				check_variable = { accepted_races^num > 4 }
				has_country_flag = accepting_a_race
			}
		}

		visible = {
			AND = {
				any_owned_state = {
					check_variable = { race = 1 }
				}
				OR = {
					NOT = {
						has_idea = ancients_primary_race
						has_idea = ancients_secondary_race
						check_variable = { accepted_races^num > 4 }
					}
				}
			}
		}

		fire_only_once = no

		cost = 300
		days_remove = 365
		days_re_enable = 365

		remove_effect = {
			add_ideas = {
				ancients_secondary_race
			}
			add_to_array = {
				array = accepted_races
				value = 1
			}
			clr_country_flag = accepting_a_race
		}

		complete_effect = {
			set_country_flag = accepting_a_race
		}

	}
